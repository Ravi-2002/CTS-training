toggleOn(e, service, i) {
  e.preventDefault();
  const { selectedDevice } = this.props;
  const services = this.state.services;
  const sfoCode = services[service].data[i].sfoCode;
  const newsfoEnabled = services[service].data[i].sfoEnabled === "Y" ? "N" : "Y";

  this.setState({ updatedServiceCode: sfoCode, successMsg: "", status: false });

  const payload = {
    mdn: selectedDevice.encryptedMtn,
    sfoCode,
    sfoEnabled: newsfoEnabled,
  };

  this.props.actions.postServices(payload);
}
