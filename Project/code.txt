import React, { useState } from "react";
import "bootstrap/dist/css/bootstrap.min.css";

const BlockServices = () => {
  const [settings, setSettings] = useState({
    videoDownloads: false,
    pictureMessaging: false,
    callerID: false,
    appDownloads: true,
    allMessaging: false,
    purchasesGifting: false,
  });

  const toggleSetting = (key) => {
    setSettings((prev) => ({ ...prev, [key]: !prev[key] }));
  };

  const options = [
    { key: "videoDownloads", label: "Block Video Downloads" },
    { key: "pictureMessaging", label: "Block Picture & Video Messaging" },
    { key: "callerID", label: "Caller ID Blocking" },
    { key: "appDownloads", label: "Block App Downloads" },
    { key: "allMessaging", label: "Block All Messaging" },
    { key: "purchasesGifting", label: "Block Purchases and Gifting" },
  ];

  return (
    <div className="container mt-4">
      <h1 className="h4 mb-1">Be in charge of who does what on your account.</h1>
      <p className="text-muted mb-4">732.609.9414</p>
      {options.map(({ key, label }) => (
        <div key={key} className="d-flex justify-content-between align-items-center border-bottom py-3">
          <div>
            <p className="mb-1 fw-medium">{label}</p>
            <a href="#" className="text-primary small">Tell Me More</a>
          </div>
          <div className="form-check form-switch">
            <input
              className="form-check-input"
              type="checkbox"
              role="switch"
              id={key}
              checked={settings[key]}
              onChange={() => toggleSetting(key)}
            />
          </div>
        </div>
      ))}
    </div>
  );
};

export default BlockServices;


The component now uses only Bootstrap for styling, including Bootstrap's switch toggles. Let me know if you want it responsive or need to integrate it with backend logic.
