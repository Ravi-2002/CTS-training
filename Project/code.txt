test("when manageTouchlessScanFFlag is false, no redirect to /success", () => {
  // Set the flag to FALSE
  reactGlobals.manageTouchlessScanFFlag = false;

  const { history } = renderWithRouter(<Submission />, {
    url: "data:image/png;base64,FLAGTEST",
  });

  fireEvent.click(screen.getByText("Submit"));

  expect(postHttpClientRequest).toHaveBeenCalledTimes(1);

  // ‚ùó NO redirect should happen
  expect(history.location.pathname).not.toBe("/success");
});
