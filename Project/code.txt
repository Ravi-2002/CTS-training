componentDidUpdate(prevProps, prevState) {
  if (
    prevProps.confirmServices !== this.props.confirmServices &&
    this.props.confirmServices?.responseInfo?.responseCode === "00"
  ) {
    const updatedServiceCode = this.state.updatedServiceCode;

    // Find and toggle it in state
    const services = { ...this.state.services };
    Object.keys(services).forEach((key) => {
      services[key].data = services[key].data.map((item) => {
        if (item.sfoCode === updatedServiceCode) {
          item.sfoEnabled = item.sfoEnabled === "Y" ? "N" : "Y";
        }
        return item;
      });
    });

    this.setState({
      services,
      successMsg: `We have successfully saved settings.`,
      status: true,
      updatedServiceCode: null
    });
  }
}
