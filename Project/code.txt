test('updates formState and resets lastToggledService when responseCode is 08 and lastToggledService is valid', () => {
  setup({
    confirmServices: {
      responseInfo: { responseCode: '08' }
    },
    lastToggledService: {
      serviceName: 'Call Blocking',
      toggledValue: false,
      sfoCode: 'CB01',
      sfoEnabled: 'N'
    },
    services: {
      General: [{ serviceName: 'Call Blocking', sfoEnabled: 'Y', sfoCode: 'CB01' }]
    }
  });

  const toggle = screen.getByTestId('Call Blocking');
  expect(toggle).toHaveTextContent('Off'); // Updated based on toggledValue: false
});
