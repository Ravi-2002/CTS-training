it('stores MDN in localStorage when MDN is provided', () => {
  // Arrange
  const mockSetItem = jest.spyOn(window.localStorage.__proto__, 'setItem');
  mockSetItem.mockImplementation(() => {});
  
  const MDN = '9876543210';

  // Act
  if (MDN) {
    localStorage.setItem('mdn', MDN);
  }

  // Assert
  expect(mockSetItem).toHaveBeenCalledWith('mdn', MDN);

  mockSetItem.mockRestore();
});
