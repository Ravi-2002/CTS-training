it("calls okClick via clickHandler when Confirm button clicked", () => {
  render(
    <CommonModal
      modalId="m9"
      message={{ ...baseMessage, button2: "Confirm" }}
      open
      close={closeMock}
      okClick={okClickMock}
    />
  );

  fireEvent.click(screen.getByText("Confirm"));
  expect(okClickMock).toHaveBeenCalled();
});

it("calls close via closeHandler when onOpenedChange(null)", () => {
  render(
    <CommonModal
      modalId="m10"
      message={baseMessage}
      open
      close={closeMock}
      okClick={okClickMock}
    />
  );

  // Get the modal wrapper
  const modal = screen.getByLabelText("number share modal");

  // Fire the onOpenedChange with null (triggers !e)
  fireEvent(modal, new CustomEvent("openedChange", { detail: null }));

  // Manually call since VDS Modal may not propagate event
  modal.props?.onOpenedChange?.(null);

  expect(closeMock).toHaveBeenCalled();
});
