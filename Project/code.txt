if (
      prevProps.confirmServices !== this.props.confirmServices &&
      this.props.confirmServices?.responseInfo?.responseCode === "00"
    ) {
      const { pendingService } = this.state;
      if (pendingService) {
        const { service, i } = pendingService;
        const { services } = this.state;
  
        services[service].data[i].sfoEnabled =
          services[service].data[i].sfoEnabled === "Y" ? "N" : "Y";
  
        this.setState({
          services,
          updatedService: services[service].data[i].sfoCode,
          successMsg: `We have successfully saved settings for "${services[service].data[i].serviceName}"`,
          pendingService: null,
        });
      }
    }
