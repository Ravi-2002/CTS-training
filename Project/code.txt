it("renders all alert messages including powerCycle and blockAgainSuccess", async () => {
  const { rerender } = render(
    <Provider store={store}>
      <BlockCallsMessages
        selectedAccordian="CM"
        selectedDevice={{ encryptedMtn: "mockMtn" }}
        blockedNumbers={[]}
        callsMessagesData={{ remainingBlockCount: 1 }}
        addBlockCallMsgRes={{
          blockedPhoneNumbers: [],
          remainingBlockCount: 0,
          successMsg: "Added successfully",
          failureMsg: "Failed to add",
          hasSuccessMsg: true,
          hasFailureMsg: true,
          powerCycle: true,
        }}
        deleteBlockCallMsgRes={{
          blockedPhoneNumbers: [],
          remainingBlockCount: 0,
          successMsg: "Deleted successfully",
          hasSuccessMsg: true,
        }}
        blockAgainResponse={{
          successMsg: "Blocked again successfully",
          failureMsg: "Block again failed",
        }}
      />
    </Provider>
  );

  // Alerts for addSuccess + failSuccess + powerCycle
  expect(await screen.findByText(/Added successfully/i)).toBeInTheDocument();
  expect(await screen.findByText(/Failed to add/i)).toBeInTheDocument();
  expect(await screen.findByText(/In order to complete your service change/i)).toBeInTheDocument();

  // Simulate blockAgainSuccess = true
  rerender(
    <Provider store={store}>
      <BlockCallsMessages
        selectedAccordian="CM"
        selectedDevice={{ encryptedMtn: "mockMtn" }}
        blockedNumbers={[]}
        callsMessagesData={{ remainingBlockCount: 1 }}
        addBlockCallMsgRes={{ blockedPhoneNumbers: [], remainingBlockCount: 0 }}
        deleteBlockCallMsgRes={{ blockedPhoneNumbers: [], remainingBlockCount: 0 }}
        blockAgainResponse={{
          successMsg: "Blocked again successfully",
          failureMsg: "Block again failed",
        }}
        state={{ blockAgainSuccess: true }}
      />
    </Provider>
  );

  expect(screen.getByText(/Blocked again successfully/i)).toBeInTheDocument();

  // Simulate blockAgainSuccess = false
  rerender(
    <Provider store={store}>
      <BlockCallsMessages
        selectedAccordian="CM"
        selectedDevice={{ encryptedMtn: "mockMtn" }}
        blockedNumbers={[]}
        callsMessagesData={{ remainingBlockCount: 1 }}
        addBlockCallMsgRes={{ blockedPhoneNumbers: [], remainingBlockCount: 0 }}
        deleteBlockCallMsgRes={{ blockedPhoneNumbers: [], remainingBlockCount: 0 }}
        blockAgainResponse={{
          successMsg: "Blocked again successfully",
          failureMsg: "Block again failed",
        }}
        state={{ blockAgainSuccess: false }}
      />
    </Provider>
  );

  expect(screen.getByText(/Block again failed/i)).toBeInTheDocument();
});
