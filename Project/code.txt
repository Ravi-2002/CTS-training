componentDidUpdate(prevProps, prevState) {
  const { services, serviceStatus } = this.props;

  if (
    serviceStatus &&
    serviceStatus.responseInfo.responseCode === "00" &&
    services &&
    Object.keys(services).length > 0 &&
    Object.keys(prevState.services).length === 0 // Ensuring we update state only when needed
  ) {
    const obj = {};
    const additionalSer = "Additional Services";
    let isAdditionalServiceAvailable = false;
    const emailDomainObj = { [BLOCK_EMAIL_DOMAIN]: { isExpand: false } };

    Object.keys(services).forEach((key) => {
      services[key].forEach((item) => {
        item.showMore = false;
      });
      if (key === additionalSer) isAdditionalServiceAvailable = true;
      obj[key] = {
        isExpand: false,
        data: services[key],
      };
    });

    const { [additionalSer]: additionalService, ...rest } = obj;

    this.setState((prevState) => {
      if (prevState.services !== obj) {
        return {
          services: isAdditionalServiceAvailable
            ? { ...emailDomainObj, ...rest, [additionalSer]: additionalService }
            : { ...emailDomainObj, ...rest },
        };
      }
      return null; // Prevents unnecessary re-renders
    });
  }
}
