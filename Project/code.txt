checkAllowedPattern = (addStr) => {
    let addPattern = /^[0-9a-zA-Z\s=,.:_-]+$/;
    let spacePattern = /^[\s=,.:_-]+$/;
    if ((addStr.length == 1) && addStr.match(spacePattern)) {
      return false;
    }
    else if (addStr.match(addPattern) || addStr == "") {
      return true;
    } else
      return false;
  }
