import React, { useState } from "react";
import { Switch } from "@headlessui/react";

const BlockServices = () => {
  const [settings, setSettings] = useState({
    videoDownloads: false,
    pictureMessaging: false,
    callerID: false,
    appDownloads: true,
    allMessaging: false,
    purchasesGifting: false,
  });

  const toggleSetting = (key) => {
    setSettings((prev) => ({ ...prev, [key]: !prev[key] }));
  };

  const options = [
    { key: "videoDownloads", label: "Block Video Downloads" },
    { key: "pictureMessaging", label: "Block Picture & Video Messaging" },
    { key: "callerID", label: "Caller ID Blocking" },
    { key: "appDownloads", label: "Block App Downloads" },
    { key: "allMessaging", label: "Block All Messaging" },
    { key: "purchasesGifting", label: "Block Purchases and Gifting" },
  ];

  return (
    <div className="p-4 max-w-md mx-auto">
      <h1 className="text-2xl font-bold mb-1">Be in charge of who does what on your account.</h1>
      <p className="mb-6 text-gray-500">732.609.9414</p>
      {options.map(({ key, label }) => (
        <div key={key} className="flex items-center justify-between border-b py-4">
          <div>
            <p className="font-medium">{label}</p>
            <a href="#" className="text-sm text-blue-600 underline">Tell Me More</a>
          </div>
          <Switch
            checked={settings[key]}
            onChange={() => toggleSetting(key)}
            className={`${settings[key] ? 'bg-green-500' : 'bg-black'} relative inline-flex h-6 w-11 items-center rounded-full`}
          >
            <span
              className={`${settings[key] ? 'translate-x-6' : 'translate-x-1'} inline-block h-4 w-4 transform rounded-full bg-white transition`}
            />
          </Switch>
        </div>
      ))}
    </div>
  );
};

export default BlockServices;
