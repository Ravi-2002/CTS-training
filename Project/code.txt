it('handles case when Content or sections[0] is missing', () => {
  const store = mockStore({
    Detail: {
      callsMessages: {
        aemResponse: null, // <-- This triggers else path
      },
    },
  });

  render(
    <Provider store={store}>
      <MemoryRouter>
        <BlockCallsEmpty />
      </MemoryRouter>
    </Provider>
  );

  // No sections, so the "else" branch is covered
  expect(screen.getByText(/Block number/i)).toBeInTheDocument();
});
it('handles Content with empty sections array', () => {
  const store = mockStore({
    Detail: {
      callsMessages: {
        aemResponse: { sections: [] }, // sections[0] is undefined
      },
    },
  });

  render(
    <Provider store={store}>
      <MemoryRouter>
        <BlockCallsEmpty />
      </MemoryRouter>
    </Provider>
  );
});
