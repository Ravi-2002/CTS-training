import React, { useState } from "react"; import "bootstrap/dist/css/bootstrap.min.css";

const BlockCallsAndMessages = () => { const [blockedNumbers, setBlockedNumbers] = useState([]); const [phoneNumber, setPhoneNumber] = useState("");

const handleAddNumber = () => { if (phoneNumber.trim() && blockedNumbers.length < 5) { setBlockedNumbers([...blockedNumbers, phoneNumber.trim()]); setPhoneNumber(""); } };

const handleRemoveNumber = (number) => { setBlockedNumbers(blockedNumbers.filter((num) => num !== number)); };

return ( <div className="container mt-4"> <h1 className="h4">Block calls and messages from up to 5 numbers.</h1> <p className="text-muted mb-4">732.609.9414</p>

{blockedNumbers.length === 0 ? (
    <>
      <div className="mb-3">
        <label htmlFor="phoneInput" className="form-label">
          Enter phone number
        </label>
        <input
          type="text"
          className="form-control"
          id="phoneInput"
          value={phoneNumber}
          onChange={(e) => setPhoneNumber(e.target.value)}
          maxLength={15}
        />
      </div>
      <button
        className="btn btn-secondary w-100"
        disabled={!phoneNumber.trim() || blockedNumbers.length >= 5}
        onClick={handleAddNumber}
      >
        Block number
      </button>
    </>
  ) : (
    <>
      <p className="fw-medium mb-3">
        You are blocking {blockedNumbers.length} of 5 numbers.
      </p>
      <ul className="list-group mb-3">
        {blockedNumbers.map((num, index) => (
          <li
            key={index}
            className="list-group-item d-flex justify-content-between align-items-center"
          >
            {num}
            <button
              className="btn btn-sm btn-outline-danger"
              onClick={() => handleRemoveNumber(num)}
            >
              &times;
            </button>
          </li>
        ))}
      </ul>
      {blockedNumbers.length < 5 && (
        <>
          <div className="mb-3">
            <input
              type="text"
              className="form-control"
              placeholder="Enter phone number"
              value={phoneNumber}
              onChange={(e) => setPhoneNumber(e.target.value)}
            />
          </div>
          <button
            className="btn btn-dark w-100"
            onClick={handleAddNumber}
            disabled={!phoneNumber.trim()}
          >
            Add a number
          </button>
        </>
      )}
    </>
  )}

  <p className="mt-4 text-muted">
    There is no charge to block.{' '}
    <a href="#" className="text-decoration-underline">
      View limitations.
    </a>
  </p>
</div>

); };

export default BlockCallsAndMessages;
