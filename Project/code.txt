it('renders service toggle and toggles it correctly', () => {
  setup({
    services: {
      Category1: [
        { serviceName: 'Service A', sfoEnabled: 'N', sfoCode: '001' }, // Initially Off
      ]
    },
    isFetching: false,
    serviceFail: false,
    confirmServices: {},
    servicesError: {}
  });

  const toggle = screen.getByTestId('test-toggle');
  expect(toggle).not.toBeChecked(); // initially off

  fireEvent.click(toggle); // toggle it

  expect(mockDispatch).toHaveBeenCalled(); // or any specific dispatched action
});
