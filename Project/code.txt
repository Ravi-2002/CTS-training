import React, { lazy, Fragment } from 'react'
import ScrollToTop from './ScrollToTop'
import { BrowserRouter, Route} from 'react-router-dom/cjs/react-router-dom.min'
const Home = lazy(() => import('./home/components'))
const DeviceDetail = lazy(() => import('./deviceDetail/components'))
const ServiceDetails =lazy(() =>import('./home/components/servicedetails'))
const ViewLimitations = lazy(() =>import('./deviceDetail/components/ViewLimitation'))
const callsmessages = lazy(() =>import('./deviceDetail/components/callsandmessages'))

const Routes = ({ reduxStore }) => {
  const routeConfig = (
    <div>
      <ScrollToTop />
      <BrowserRouter>
      <Route exact path="/blockService" component={Home} />
      <Route exact path="/service-details/:serviceName" component = {ServiceDetails} />
      <Route exact path="/call-message-blocking" component = {callsmessages} />
      <Route exact path="/callsandmessages" component={DeviceDetail} />
      <Route exact path="/viewlimitation" component={ViewLimitations} />
      </BrowserRouter>
    </div>
  )
  return <Fragment>{routeConfig}</Fragment>
}

export default Routes
