import React from 'react';
import { render, screen, fireEvent } from '@testing-library/react';
import { Provider, useDispatch, useSelector } from 'react-redux';
import { Router } from 'react-router-dom';
import { createMemoryHistory } from 'history';
import configureStore from 'redux-mock-store';
import '@testing-library/jest-dom';
import BlockServices from '../../components/index';
import servicesReducer from '../../redux/reducers/servicesReducer'; // adjust path if needed
import * as actions from '../../redux/actions/blockServicesActions'; // adjust path

// ---------- MOCKS ----------
jest.mock('react-redux', () => ({
  ...jest.requireActual('react-redux'),
  useDispatch: jest.fn(),
  useSelector: jest.fn(),
}));

const mockStore = configureStore([]);
const mockDispatch = jest.fn();

// ---------- COMPONENT TEST ----------
describe('BlockServices Component', () => {
  let store;
  let history;

  beforeEach(() => {
    useDispatch.mockReturnValue(mockDispatch);
  });

  const setup = (selectorData) => {
    store = mockStore({});
    history = createMemoryHistory();
    history.push('/block-services?mdn=1234567890');

    useSelector.mockImplementation((cb) => cb({ Home: selectorData }));

    render(
      <Provider store={store}>
        <Router history={history}>
          <BlockServices />
        </Router>
      </Provider>
    );

    return { store, history };
  };

  afterEach(() => {
    jest.clearAllMocks();
  });

  it('renders and toggles service correctly', () => {
    setup({
      services: {
        Category1: [
          { serviceName: 'Service A', sfoEnabled: 'Y', sfoCode: '001' },
          { serviceName: 'Service B', sfoEnabled: 'N', sfoCode: '002' },
        ],
      },
      isFetching: false,
      serviceFail: false,
      confirmServices: {},
      servicesError: {},
    });

    expect(screen.getByText(/Be in charge of who does what/i)).toBeInTheDocument();
    expect(screen.getByText('Service A')).toBeInTheDocument();
    expect(screen.getByText('Service B')).toBeInTheDocument();

    fireEvent.click(screen.getAllByRole('checkbox')[1]);
    expect(mockDispatch).toHaveBeenCalled();
  });

  it('shows success notification when confirmServices responseCode is 00', () => {
    setup({
      services: { Category1: [] },
      isFetching: false,
      serviceFail: false,
      confirmServices: { responseInfo: { responseCode: '00' } },
      servicesError: {},
    });

    expect(screen.getByText(/Service updated successfully/i)).toBeInTheDocument();
  });

  it('shows error notification when responseCode is 99', () => {
    setup({
      services: { Category1: [] },
      isFetching: false,
      serviceFail: false,
      confirmServices: {},
      servicesError: {
        responseInfo: { responseCode: '99' },
        libErrorInfo: {
          internalErrors: [{ errorMessage: 'Sample error' }],
        },
      },
    });

    expect(screen.getByText(/Error/i)).toBeInTheDocument();
  });

  it('renders failure message if service load fails', () => {
    setup({
      services: {},
      isFetching: false,
      serviceFail: true,
      confirmServices: {},
      servicesError: {},
    });

    expect(screen.getByText(/Failed to fetch services/i)).toBeInTheDocument();
  });

  it('navigates to service details on TextLink click', () => {
    const { history } = setup({
      services: {
        Category1: [{ serviceName: 'Service A', sfoEnabled: 'Y', sfoCode: '001' }],
      },
      isFetching: false,
      serviceFail: false,
      confirmServices: {},
      servicesError: {},
    });

    fireEvent.click(screen.getByText(/Tell me more/i));
    expect(history.location.pathname).toContain('/service-details/Service A');
  });
});

// ---------- REDUCER TEST ----------
describe('servicesReducer', () => {
  const initialState = {
    isFetching: false,
    services: {},
    confirmServices: {},
    servicesError: {},
    serviceErrorFlag: false,
    confirmServiceFlag: false,
    failServices: {},
    getServices: false,
    serviceFail: false,
  };

  it('should return initial state', () => {
    expect(servicesReducer(undefined, {})).toEqual(initialState);
  });

  it('should handle GET_SERVICES_BEGIN', () => {
    const action = { type: actions.GET_SERVICES_BEGIN };
    const expected = {
      ...initialState,
      isFetching: true,
    };
    expect(servicesReducer(initialState, action)).toEqual(expected);
  });

  it('should handle GET_SERVICES_SUCCESS', () => {
    const mockServices = { Category1: [] };
    const action = {
      type: actions.GET_SERVICES_SUCCESS,
      services: mockServices,
      serviceStatus: {},
    };
    const expected = {
      ...initialState,
      services: mockServices,
      serviceStatus: {},
      getServices: true,
      isFetching: false,
    };
    expect(servicesReducer(initialState, action)).toEqual(expected);
  });

  it('should handle GET_SERVICES_FAIL', () => {
    const action = {
      type: actions.GET_SERVICES_FAIL,
      failServices: { error: 'Failed' },
    };
    const expected = {
      ...initialState,
      serviceFail: true,
      isFetching: false,
      failServices: { error: 'Failed' },
    };
    expect(servicesReducer(initialState, action)).toEqual(expected);
  });

  it('should handle POST_SERVICES_SUCCESS', () => {
    const action = {
      type: actions.POST_SERVICES_SUCCESS,
      confirmServices: { message: 'success' },
    };
    const expected = {
      ...initialState,
      confirmServices: { message: 'success' },
      confirmServiceFlag: true,
      isFetching: false,
    };
    expect(servicesReducer(initialState, action)).toEqual(expected);
  });

  it('should handle POST_SERVICES_FAIL', () => {
    const action = {
      type: actions.POST_SERVICES_FAIL,
      servicesError: { message: 'error' },
    };
    const expected = {
      ...initialState,
      servicesError: { message: 'error' },
      serviceErrorFlag: true,
      isFetching: false,
    };
    expect(servicesReducer(initialState, action)).toEqual(expected);
  });
});

// ---------- ACTION CREATOR TEST ----------
describe('blockServicesActions', () => {
  it('should create GET_SERVICES_BEGIN action', () => {
    expect(actions.getServicesBegin()).toEqual({ type: actions.GET_SERVICES_BEGIN });
  });

  it('should create GET_SERVICES_SUCCESS action', () => {
    const mockResp = {
      body: {
        sections: [{ sections: [{ data: { serviceDetailsMap: { a: [] } } }] }],
      },
    };
    const expected = {
      type: actions.GET_SERVICES_SUCCESS,
      services: { a: [] },
      serviceStatus: mockResp,
    };
    expect(actions.getServicesSuccess(mockResp)).toEqual(expected);
  });

  it('should create GET_SERVICES_FAIL action', () => {
    const response = { body: { error: true } };
    const expected = {
      type: actions.GET_SERVICES_FAIL,
      failServices: response.body,
    };
    expect(actions.getServicesFail(response)).toEqual(expected);
  });

  it('should create POST_SERVICES_BEGIN action', () => {
    expect(actions.postServicesBegin()).toEqual({ type: actions.POST_SERVICES_BEGIN });
  });

  it('should create POST_SERVICES_SUCCESS action', () => {
    const response = { success: true };
    expect(actions.postServicesSuccess(response)).toEqual({
      type: actions.POST_SERVICES_SUCCESS,
      confirmServices: response,
    });
  });

  it('should create POST_SERVICES_FAIL action', () => {
    const response = { error: true };
    expect(actions.postServicesFail(response)).toEqual({
      type: actions.POST_SERVICES_FAIL,
      servicesError: response,
    });
  });
});
