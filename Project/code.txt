import { render } from '@testing-library/react';
import { Provider } from 'react-redux';
import configureStore from 'redux-mock-store';
import { MemoryRouter } from 'react-router-dom';
import BlockCallsEmpty from '../BlockCallsEmpty';

const mockStore = configureStore([]);

describe('BlockCallsEmpty', () => {
  it('covers MDN truthy branch and stores MDN in localStorage', () => {
    const store = mockStore({
      Detail: {
        callsMessages: {
          blockedNumbers: [],
          addBlockCallMsg: {},
          isFetching: false,
        },
      },
    });

    // Spy on localStorage
    const setItemSpy = jest.spyOn(Storage.prototype, 'setItem');

    render(
      <Provider store={store}>
        <MemoryRouter
          initialEntries={['/callsandMessages?MDN=5556667777']}
        >
          <BlockCallsEmpty />
        </MemoryRouter>
      </Provider>
    );

    expect(setItemSpy).toHaveBeenCalledWith('mdn', '5556667777');
  });
});
