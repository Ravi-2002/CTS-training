componentDidUpdate(prevProps) {
  const { confirmServices, confirmServiceFlag, services } = this.props;

  // Update toggle switch when confirmService is received
  if (
    prevProps.confirmServices !== confirmServices &&
    confirmServiceFlag &&
    confirmServices?.responseInfo?.responseCode === "00"
  ) {
    const updatedServiceCode = confirmServices?.sfoCode;
    const updatedServiceName = confirmServices?.serviceName;

    if (updatedServiceCode) {
      const servicesCopy = { ...this.state.services };

      Object.keys(servicesCopy).forEach((key) => {
        if (Array.isArray(servicesCopy[key]?.data)) {
          servicesCopy[key].data = servicesCopy[key].data.map((item) => {
            if (item.sfoCode === updatedServiceCode) {
              return {
                ...item,
                sfoEnabled: item.sfoEnabled === "Y" ? "N" : "Y"
              };
            }
            return item;
          });
        }
      });

      this.setState({
        services: servicesCopy,
        status: true,
        successMsg: `We have successfully saved settings for "${updatedServiceName || updatedServiceCode}"`
      });
    }
  }

  // Keep your existing service loading logic here unchanged...
}
