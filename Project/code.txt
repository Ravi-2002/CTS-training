it('updates form state using lastToggledService when confirmServices responseCode is 08', () => {
    setup({
      services: {
        Category1: [{ serviceName: 'Service A', sfoEnabled: 'Y', sfoCode: '001' }],
      },
      isFetching: false,
      serviceFail: false,
      confirmServices: {
        responseInfo: { responseCode: '08' },
      },
      lastToggledService: {
        serviceName: 'Service A',
        sfoCode: '001',
        toggledValue: false,
        sfoEnabled: 'N',
      },
      servicesError: {},
    });

    const toggle = screen.getByTestId('Service A');
    expect(toggle).toHaveTextContent('Off'); // toggledValue was false
  });
