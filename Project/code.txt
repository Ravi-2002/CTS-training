describe("populateTransactionType additional coverage", () => {

  it("should set NOTHANKS when optOut = 'Y' and destinationFlow = 'SCAN_ID'", () => {
    const values = {};
    const optOut = "Y";
    const reactGlobals = { destinationFlow: "SCAN_ID" };

    populateTransactionType(values, optOut, reactGlobals);

    expect(values.customerOpted).toBe("N");
    expect(values.transactionType).toBe("MVO_ID_ADT_AUTH_NOTHANKS");
  });

  it("should set SMART_LINK_NOTHANKS when optOut = 'Y' but destinationFlow != 'SCAN_ID'", () => {
    const values = {};
    const optOut = "Y";
    const reactGlobals = { destinationFlow: "OTHER" };

    populateTransactionType(values, optOut, reactGlobals);

    expect(values.customerOpted).toBe("N");
    expect(values.transactionType).toBe("MVO_ID_SMART_LINK_NOTHANKS");
  });

  it("should set TOUCHLESS when optOut != 'Y' and destinationFlow = 'SCAN_ID'", () => {
    const values = {};
    const optOut = "N";    // or undefined or null
    const reactGlobals = { destinationFlow: "SCAN_ID" };

    populateTransactionType(values, optOut, reactGlobals);

    expect(values.customerOpted).toBe("Y");
    expect(values.transactionType).toBe("MVO_ID_ADT_AUTH_TOUCHLESS");
  });

  it("should set SMART_LINK_TOUCHLESS when optOut != 'Y' and destinationFlow != 'SCAN_ID'", () => {
    const values = {};
    const optOut = "N";
    const reactGlobals = { destinationFlow: "OTHER" };

    populateTransactionType(values, optOut, reactGlobals);

    expect(values.customerOpted).toBe("Y");
    expect(values.transactionType).toBe("MVO_ID_SMART_LINK_TOUCHLESS");
  });

});
