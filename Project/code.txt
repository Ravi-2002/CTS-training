it("covers MDN truthy branch and stores MDN from query string", () => {
  useDispatch.mockReturnValue(mockDispatch);
  useSelector.mockImplementation(cb =>
    cb({
      Detail: {
        blockedNumbers: [],
        addBlockCallMsg: {},
        isFetching: false,
        callsMessages: {
          aemResponse: { sections: [{ sections: [{ data: { mtn: "7703463247" } }] }] }
        }
      }
    })
  );

  useLocation.mockReturnValue({
    state: {}, // no mtn in state
    search: "?mdn=5556667777"
  });

  const store = mockStore({});
  const history = createMemoryHistory();

  const setItemSpy = jest.spyOn(Storage.prototype, "setItem");

  render(
    <Provider store={store}>
      <Router history={history}>
        <BlockCallsEmpty history={history} />
      </Router>
    </Provider>
  );

  expect(setItemSpy).toHaveBeenCalledWith("mdn", "5556667777");
});
